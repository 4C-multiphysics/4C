\long\def\@makecaption#1#2{%
  \vskip-0.65\abovecaptionskip
  \sbox\@tempboxa{\textbf{#1:} #2}%
  \ifdim \wd\@tempboxa >\hsize
    \textbf{#1:} #2\par
  \else

  \fi
  \vskip\belowcaptionskip}
