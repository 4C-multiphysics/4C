\subsubsection{Arc-length control}
\subitempart{Control Flag}
\begin{quote}
\cod{NEWTONRAPHSO} \cod{Arc\_Control}\\
\end{quote}

\subitempart{Variants}
\begin{quote}
\cod{IARC} \cod{Yes} $|$ \cod{No}\\ 
\cod{ARCSCL} \cod{0} $|$ \cod{1}\\
By default cylindrical method is used, only if \cod{IARC} \cod{Yes}
AND \cod{ARCSCL} \cod{1} spherical method is applied (cf. Crisfield)\\
\end{quote}

\subitempart{Options}
\begin{quote}
\cod{SIGNCHCSP} \cod{Yes} $|$ \cod{No}

 ``flags on a better control of
sign for the predictor depending on the sign of the current stiffness,
i.e. to improve handling of limit points''
\end{quote}
